# Requirements for Streamlit RAG Chatbot with Gemini API and ChromaDB

# Streamlit framework
streamlit

# Google Gemini API SDK
google-genai

# Vector Store (ChromaDB) and its dependencies
chromadb
pysqlite3-binary  # CRITICAL for Streamlit Cloud deployment with ChromaDB

# Components from the LangChain ecosystem
langchain-text-splitters
langchain-core # Often needed for traceable (LangSmith) components
langsmith # For tracing/monitoring

# Embedding model dependency
sentence-transformers

# Document Loaders/Parsers
pypdf # For PDF processing
requests # For fetching GitHub URLs
# New dependencies for additional file types
pandas # For CSV/Excel processing (though only CSV is requested)
beautifulsoup4 # For HTML/XML parsing
lxml # Generally a good dependency for robust XML/HTML parsing

# New dependency for Text-to-Speech (TTS) functionality
edge-tts
